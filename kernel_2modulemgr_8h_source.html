35 /*
36  * Assigning the following macro to the variable sceKernelPreloadModuleInhibit with the OR operator inhibit preloading that module.
37  *
38  * Example
39  * <code>
40  * // Inhibit preload SceLibc and SceShellSvc.
41  * int sceKernelPreloadModuleInhibit = SCE_KERNEL_PRELOAD_INHIBIT_LIBC | SCE_KERNEL_PRELOAD_INHIBIT_LIBSHELLSVC;
42  * </code>
43  *
44  * And these are only valid for modules in the process image, preload is not inhibited even if specified for modules to be loaded later.
45  *
46  * WARNING
47  * If SceLibNet etc. is loaded without SceShellSvc etc. loaded, an unintended system crash will occur.
48  */
49 typedef enum SceKernelPreloadInhibit {
